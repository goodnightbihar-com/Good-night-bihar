Sample Google Apps Script to receive POSTs and email to your Gmail (deploy as Web App)

// Google Apps Script server-side code (Code.gs) function doPost(e){ try{ var data = JSON.parse(e.postData.contents); var body = Object.keys(data).map(k => k+": "+data[k]).join('\n'); MailApp.sendEmail('YOUR_EMAIL@gmail.com','Booster form submission',body); return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); }catch(err){ return ContentService.createTextOutput(JSON.stringify({status:'error', message:err.message})).setMimeType(ContentService.MimeType.JSON); } }
